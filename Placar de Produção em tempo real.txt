const addSeparators = function (nStr, thousandsSep, decimalSep) {
    const x = String(nStr).split(".");
    let x1 = x[0];
    const x2 = x.length > 1 ? decimalSep + x[1] : "";
    const rgx = /(\d+)(\d{3})/;
    while (rgx.test(x1)) {
        x1 = x1.replace(rgx, `$1${thousandsSep}$2`);
    }
    return x1 + x2;
};

const numberFormat = function (optsIn) {
    const defaults = {
        digitsAfterDecimal: 2,
        scaler: 1,
        thousandsSep: ",",
        decimalSep: ".",
        prefix: "",
        suffix: "",
    };
    const opts = Object.assign({}, defaults, optsIn);
    return function (x) {
        if (isNaN(x) || !isFinite(x)) {
            return "";
        }
        const result = addSeparators(
            (opts.scaler * x).toFixed(opts.digitsAfterDecimal),
            opts.thousandsSep,
            opts.decimalSep
        );
        return `${opts.prefix}${result}${opts.suffix}`;
    };
};

        async carregarTotaisUltimos7Dias() {
            var thisVM = this;

            thisVM.dadosUltimos7Dias = [];

            const dataBase = thisVM.dataSelecionadaFormatted || thisVM.formatDateToString(new Date());

            const datas = thisVM.gerarDatasUltimos7Dias(dataBaseObj);

                                const totais = snapshot.val();

                                        valorTotalSeparacao: 0,

                                        qtdSeparacoes: 0
                                    },
                                    temDados: false
                                });

                                data: data,
                                dataFormatada: thisVM.formatarDataParaExibicao(data),
                                totais: {

                                    valorTotalSeparacao: 0,

                                    qtdSeparacoes: 0
                                },
                                temDados: false,
                                erro: true
                            });
                        });
                });

                promises.push(promise);
            });

        async carregarTotaisDoMes(anoMes) {
            var thisVM = this;

            try {
                // Buscar todos os dias do mês no Analytics
                const firebasePath = `${thisVM.$route.params.time}/Analytics/TabelaProducaoDiaria`;
                const snapshot = await firebase.database().ref(firebasePath).once('value');

                if (!snapshot.exists()) {
                    return { qtd: 0, valor: 0, custoMedio: 0 };
                }

                let totalQtdMes = 0;
                let totalValorMes = 0;

                snapshot.forEach((diaSnapshot) => {
                    const data = diaSnapshot.key;

                    // Verificar se a data pertence ao mês selecionado
                    if (data.startsWith(anoMes)) {
                        const totaisDia = diaSnapshot.child('totais').val();
                        if (totaisDia) {
                            const pesoDia = totaisDia.pesoKg || totaisDia.pesoTotalKg || 0;
                            const valorDiaCent = totaisDia.valorTotalCent || totaisDia.valorTotalSeparacao || 0;
                            const valorDia = valorDiaCent / (totaisDia.valorTotalCent ? 100 : 1);

                            totalQtdMes += pesoDia;
                            totalValorMes += valorDia;
                        }
                    }
                });

                return {
                    qtd: totalQtdMes,
                    valor: totalValorMes,
                    custoMedio: totalQtdMes > 0 ? totalValorMes / totalQtdMes : 0
                };

            } catch (error) {
                alert('Erro ao carregar totais do mês:' + error);
                return { qtd: 0, valor: 0, custoMedio: 0 };
            }
        },

        async carregarTotaisDoDia(data) {
            var thisVM = this;

            const firebasePath = `${thisVM.$route.params.time}/Analytics/TabelaProducaoDiaria/${data}/totais`;

            try {

                    const valorTotal = valorTotalCent / (totaisDia.valorTotalCent ? 100 : 1);

                        qtd: pesoTotal,
                        valor: valorTotal,
                        custoMedio: pesoTotal > 0 ? valorTotal / pesoTotal : 0
                    };
                } else {
                    return { qtd: 0, valor: 0, custoMedio: 0 };
                }

            } catch (error) {
                alert('Erro ao carregar totais do dia:' + error);
                return { qtd: 0, valor: 0, custoMedio: 0 };
            }
        },

        processarTotaisSemana(dadosDias) {
            var thisVM = this;

            let totalQtdSemana = 0;
            let totalValorSemana = 0;

            dadosDias.forEach(dia => {
                if (dia.temDados) {
                    totalQtdSemana += dia.totais.pesoKg || dia.totais.pesoTotalKg || 0;

                    const valorDia = dia.totais.valorTotalCent || dia.totais.valorTotalSeparacao || 0;
                    totalValorSemana += valorDia / (dia.totais.valorTotalCent ? 100 : 1);
                }
            });

            thisVM.totaisSemana = {
                qtd: totalQtdSemana,
                valor: totalValorSemana,
                custoMedio: totalQtdSemana > 0 ? totalValorSemana / totalQtdSemana : 0
            };
        },

        gerarDatasUltimos7Dias(dataBase) {
            const datas = [];
            const base = new Date(dataBase);

            for (let i = 6; i >= 0; i--) {
                const data = new Date(base);
                data.setDate(base.getDate() - i);
                datas.push(this.formatDateToString(data));
            }

            return datas;
        },

        formatarDataParaExibicao(dataString) {
            if (!dataString) return '';

            try {
                const data = new Date(dataString + 'T12:00:00');
                return data.toLocaleDateString('pt-BR', {
                    weekday: 'long',
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                });
            } catch (e) {
                alert("Erro ao formatar data:", e);
                return dataString;
            }
        },

            } catch (e) {

        formatDateToString(date) {
            if (!date) return '';

            // Se já for string no formato YYYY-MM-DD, retornar diretamente
            if (typeof date === 'string' && date.match(/^\d{4}-\d{2}-\d{2}$/)) {
                return date;
            }

            // Se for objeto Date, formatar
            const dateObj = new Date(date);
            const year = dateObj.getFullYear();
            const month = String(dateObj.getMonth() + 1).padStart(2, '0');
            const day = String(dateObj.getDate()).padStart(2, '0');
            return `${year}-${month}-${day}`;
        },

        processarDadosTabelaDiaria(dadosDiarios, dataString) {
            var thisVM = this;

            thisVM.rankingColaboradores = [];
            thisVM.separacoes = [];
            thisVM.colaboradoresDisponiveis = [];
            thisVM.produtosDisponiveis = [];

            if (dataString === undefined || dataString == null) {
                dataString = this.formatDateToString(new Date());
            }

            if (!dadosDiarios.itens) {
                thisVM.$buefy.toast.open({
                    message: 'Nenhum dado de produção encontrado para esta data',
                    type: 'is-warning',
                    position: 'is-top'
                });
                return;
            }

            Object.keys(dadosDiarios.itens).forEach((chaveItem, index) => {
                const item = dadosDiarios.itens[chaveItem];

                if (!item || (!item.idColaborador && !item.nomeColaborador)) {
                    alert('Item inválido na tabela diária:' + item);
                    return;
                }

                const valorPorKg = (item.valorPorKgCent || item.valorPorKg || 0) / (item.valorPorKgCent ? 100 : 1);
                const valorTotal = (item.valorTotalCent || item.valorTotal || 0) / (item.valorTotalCent ? 100 : 1);
                const valorPorPessoa = (item.valorPorPessoaCent || item.valorPorPessoa || 0) / (item.valorPorPessoaCent ? 100 : 1);

                const idColaborador = item.idColaborador || `colab_${item.nomeColaborador.replace(/\s+/g, '_').toLowerCase()}`;
                const idProduto = item.idProduto || `produto_${item.material.replace(/\s+/g, '_').toLowerCase()}`;

                const separacao = {
                    idSeparacao: `tabela_diaria_${idColaborador}_${index}_${Date.now()}`,
                    idColaborador: idColaborador,
                    nomeColaborador: item.nomeColaborador || 'Não informado',
                    codigoColaborador: item.codigoColaborador || '',
                    idProduto: idProduto,
                    nomeProduto: item.material || item.nomeProduto || 'Não informado',
                    quantidade: item.pesoKg || item.quantidade || 0,
                    dataSeparacao: item.dataSeparacao,
                    valorSeparacaoPorSeparador: valorTotal,
                    custoSeparacaoPorKG: valorPorKg,
                    precoVenda: item.precoVenda || 0,
                    precoCompra: item.precoCompra || 0,
                    categoriaProduto: item.categoriaProduto || '',
                    valorVendaTotalPorSeparador: item.valorVendaTotal || 0,
                    valorCompraTotalPorSeparador: item.valorCompraTotal || 0,
                    valorGanhoTotalPorSeparador: item.valorGanhoTotal || 0,
                    NomeEquipe: item.equipe || item.NomeEquipe || '',
                    idPesagem: item.idPesagem || '',
                    totalPesagem: item.pesoKg || item.totalPesagem || 0,
                    estorno: item.estorno || false,
                    estornado: item.estornado || false,
                    finalizada: item.finalizada !== undefined ? item.finalizada : true,
                    horaInicio: item.horaInicio || '',
                    horaTermino: item.horaTermino || '',
                    qtdPessoasEquipe: item.qtdPessoasEquipe || 0,
                    valorPorPessoa: valorPorPessoa
                };

                thisVM.separacoes.push(separacao);

                let colaboradorExistente = thisVM.rankingColaboradores.find(c => c.id === idColaborador);
                if (!colaboradorExistente) {
                    thisVM.rankingColaboradores.push({
                        id: idColaborador,
                        nome: item.nomeColaborador || 'Não informado',
                        volumeTotalProducao: item.pesoKg || item.quantidade || 0,
                        valorTotalProducao: valorTotal,
                        Produtos: [{
                            nome: item.material || item.nomeProduto || 'Não informado',
                            volume: item.pesoKg || item.quantidade || 0,
                            valorSeparacao: valorTotal
                        }]
                    });
                } else {
                    colaboradorExistente.volumeTotalProducao += item.pesoKg || item.quantidade || 0;
                    colaboradorExistente.valorTotalProducao += valorTotal;

                    const produtoExistente = colaboradorExistente.Produtos.find(p => p.nome === (item.material || item.nomeProduto));
                    if (produtoExistente) {
                        produtoExistente.volume += item.pesoKg || item.quantidade || 0;
                        produtoExistente.valorSeparacao += valorTotal;
                    } else {
                        colaboradorExistente.Produtos.push({
                            nome: item.material || item.nomeProduto || 'Não informado',
                            volume: item.pesoKg || item.quantidade || 0,
                            valorSeparacao: valorTotal
                        });
                    }
                }

                let foundColaborador = thisVM.colaboradoresDisponiveis.find(c => c.id === idColaborador);
                if (!foundColaborador) {
                    thisVM.colaboradoresDisponiveis.push({
                        id: idColaborador,
                        nome: item.nomeColaborador || 'Não informado',
                        codigo: item.codigoColaborador || ''
                    });
                }

                let foundProduto = thisVM.produtosDisponiveis.find(p => p.id === idProduto);
                if (!foundProduto) {
                    thisVM.produtosDisponiveis.push({
                        id: idProduto,
                        nome: item.material || item.nomeProduto || 'Não informado',
                        codigo: item.codigoProduto || ''
                    });
                }
            });

            thisVM.rankingColaboradores.sort((a, b) => b.volumeTotalProducao - a.volumeTotalProducao);
            thisVM.colaboradoresDisponiveis.sort((a, b) => a.nome.localeCompare(b.nome));
            thisVM.produtosDisponiveis.sort((a, b) => a.nome.localeCompare(b.nome));

            thisVM.generatePivot();

            if (thisVM.separacoes.length > 0) {
                thisVM.$buefy.toast.open({
                    message: `${thisVM.separacoes.length} registros de produção carregados`,
                    type: 'is-success',
                    position: 'is-top',
                    duration: 3000
                });
            }
        },

        startRealtimeDataLoading() {
            var thisVM = this;

            if (!thisVM.selectedFicha) return;

            if (thisVM.dataListener && typeof thisVM.dataListener === 'function') {
                thisVM.dataListener();
            }

            if (thisVM.analyticsListener && typeof thisVM.analyticsListener === 'function') {
                thisVM.analyticsListener();
            }

            if (thisVM.firebaseRef) {
                thisVM.firebaseRef.off();
            }

            thisVM.dataListener = null;
            thisVM.analyticsListener = null;
            thisVM.firebaseRef = null;

            thisVM.$store.commit("startLoading");
            thisVM.realtimeConnected = false;

            const hoje = thisVM.formatDateToString(new Date());

            if (!firebase.database) {
                thisVM.$store.commit("stopLoading");
                thisVM.$buefy.toast.open({
                    message: "Erro: Firebase não inicializado",
                    type: 'is-danger',
                    position: 'is-top'
                });
                return;
            }

            const firebasePath = `${thisVM.$route.params.time}/Analytics/TabelaProducaoDiaria/${hoje}`;

            thisVM.firebaseRef = firebase.database().ref(firebasePath);

            thisVM.analyticsListener = thisVM.firebaseRef.on('value', (snapshot) => {
                try {
                    if (!snapshot) {
                        thisVM.realtimeConnected = false;
                        thisVM.clearData();
                        thisVM.$store.commit("stopLoading");
                        return;
                    }

                    thisVM.realtimeConnected = true;

                    if (snapshot.exists() && snapshot.val()) {
                        const dadosDiarios = snapshot.val();
                        thisVM.processarDadosTabelaDiaria(dadosDiarios);
                    } else {
                        thisVM.clearData();
                    }

                    thisVM.$store.commit("stopLoading");
                } catch (error) {
                    thisVM.realtimeConnected = false;
                    thisVM.$store.commit("stopLoading");
                    thisVM.$buefy.toast.open({
                        message: "Erro ao processar dados da tabela diária: " + error.message,
                        type: 'is-danger',
                        position: 'is-top'
                    });
                }
            }, (error) => {
                thisVM.realtimeConnected = false;
                thisVM.$store.commit("stopLoading");
                thisVM.$buefy.toast.open({
                    message: "Erro ao carregar dados em tempo real: " + error.message,
                    type: 'is-danger',
                    position: 'is-top'
                });
            });

            thisVM.carregarTotaisUltimos7Dias();
        },

        filterSeparacoesColaboradoresProdutos(separacoes) {
            var thisVM = this;
            var separacoesReturn = [];
            for (let index = 0; index < Object.keys(separacoes).length; index++) {
                const element = separacoes[Object.keys(separacoes)[index]];

                let foundColaborador = thisVM.colaboradoresDisponiveis.find((c) => {
                    return c.id == element.idColaborador;
                });

                if (foundColaborador == null) {
                    thisVM.colaboradoresDisponiveis.push({
                        id: element.idColaborador,
                        nome: element.nomeColaborador,
                        codigo: element.codigoColaborador,
                    });
                }
                let foundProduto = thisVM.produtosDisponiveis.find((p) => {
                    return p.id == element.idProduto;
                });
                if (foundProduto == null) {
                    thisVM.produtosDisponiveis.push({
                        id: element.idProduto,
                        nome: element.nomeProduto,
                        codigo: element.codigoProduto,
                    });
                }
                separacoesReturn.unshift(element);
            }

            separacoesReturn = separacoesReturn.filter(
                (separacao) =>
                    (thisVM.FilterDataInicioSeparacao == null ||
                        (thisVM.FilterDataInicioSeparacao != null &&
                            DateTime.fromJSDate(new Date(separacao.dataSeparacao)) >=
                            DateTime.fromJSDate(thisVM.FilterDataInicioEntrada))) &&
                    (thisVM.FilterDataFimSeparacao == null ||
                        (thisVM.FilterDataFimSeparacao != null &&
                            DateTime.fromJSDate(new Date(separacao.dataSeparacao)) <=
                            DateTime.fromJSDate(thisVM.FilterDataFimSeparacao)))
            );

            return separacoesReturn;
        },

        totaisGeral() {
            let thisVM = this;

            // Soma das quantidades (kg) de dia, semana e mês
            const qtdTotal = thisVM.totaisPeriodo.dia.qtd +
                thisVM.totaisPeriodo.semana.qtd +

                valor: valorTotal,
